<script>
export default {
    props: ['question', 'question_number', 'modelValue'],
    data() {
        return{
        likert_options: [
            'Discordo totalmente',
            'Discordo parcialmente',
            'Não sei/Não se aplica',
            'Concordo parcialmente',
            'Concordo totalmente'
        ]
        }
    },
    methods: {
        updateAnswer(value) {
            return {
                ...this.modelValue,
                content: value
            }
        },
        updateLikertAnswer(value, index) {
            return {
                ...this.modelValue,
                likert_answers_attributes: 
                    this.modelValue.likert_answers_attributes.map((answer, idx) => {
                        if (index == idx)
                            return {
                                ...answer,
                                content: value
                            }
                        else
                            return answer
                    })
            }
        }
    }
}
</script>
